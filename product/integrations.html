API Integration Example 
// Initialize Fyndo SDK
const fyndo = new FyndoAPI({
  apiKey: 'your-api-key',
  endpoint: 'https://api.fyndo.com'
});

// Search across integrated platforms
const results = await fyndo.search({
  query: 'Q4 financial reports',
  sources: ['sharepoint', 'gdrive', 'confluence'],
  filters: {
    dateRange: 'last-3-months',
    fileType: ['pdf', 'xlsx']
  }
});

// Process search results
results.forEach(item => {
  console.log(`Found: ${item.title}`);
  console.log(`Source: ${item.source}`);
  console.log(`Relevance: ${item.score}`);
});
